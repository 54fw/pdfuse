\section{Motivation}
% =============================================================================

\begin{frame}{Why Do We Need Another PDF Tool?}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \begin{alertblock}{Pain Points with Existing Tools}
        \begin{itemize}
          \item \textbf{Cloud services} (Smallpdf, ILovePDF)\\
                {\small Upload sensitive docs to unknown servers}
          \vspace{4pt}
          \item \textbf{Paid software} (Adobe Acrobat)\\
                {\small \$20+/month for tasks that should be free}
          \vspace{4pt}
          \item \textbf{Legacy CLI tools} (pdftk, ghostscript)\\
                {\small Cryptic syntax, no progress feedback}
          \vspace{4pt}
          \item \textbf{No pipeline support}\\
                {\small Cannot chain: compress then watermark then split}
        \end{itemize}
      \end{alertblock}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{block}{\texttt{pdfuse} solves this}
        \begin{itemize}
          \item[\cmark] \textbf{100\% local} --- nothing leaves your machine
          \vspace{4pt}
          \item[\cmark] \textbf{Free \& open-source} --- MIT license
          \vspace{4pt}
          \item[\cmark] \textbf{Beautiful CLI} --- \texttt{rich} progress
                        bars and colored errors
          \vspace{4pt}
          \item[\cmark] \textbf{Batch \& pipeline} --- process entire
                        folders, chain steps in YAML
        \end{itemize}
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Feature Overview}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \begin{block}{Commands}
        \renewcommand{\arraystretch}{1.25}
        \begin{tabular}{ll}
          \texttt{merge}     & Combine multiple PDFs into one \\
          \texttt{split}     & Extract a page range \\
          \texttt{convert}   & Images / Office docs to PDF \\
          \texttt{compress}  & Lossless deflate compression \\
          \texttt{rotate}    & Rotate pages 90/180/270 deg \\
          \texttt{watermark} & Text or PDF stamp overlay \\
          \texttt{reorder}   & Rearrange or duplicate pages \\
          \texttt{batch}     & YAML multi-step pipeline \\
        \end{tabular}
      \end{block}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{block}{Every Command Supports}
        \begin{itemize}
          \item \texttt{--folder DIR} --- process an entire directory
          \item \texttt{--recursive} --- include subdirectories
          \item \texttt{--workers N} --- parallel processing
          \item \texttt{--pattern GLOB} --- file filter
          \item \texttt{-o / --output} --- consistent output flag
          \item \texttt{rich} progress bars \& colored errors
        \end{itemize}
      \end{block}
      \vspace{6pt}
      \begin{block}{Batch Pipeline}
        \begin{itemize}
          \item Chain any steps in a YAML file
          \item Temp files auto-cleaned; original never modified
        \end{itemize}
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}[fragile]{Flags \& Options}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \begin{block}{Shared Flags (Every Command)}
        \small
        \renewcommand{\arraystretch}{1.2}
        \begin{tabular}{ll}
          \texttt{-o / --output}   & Output path or directory \\
          \texttt{--folder DIR}    & Process entire directory \\
          \texttt{--recursive}     & Recurse into subdirectories \\
          \texttt{--workers N}     & Parallel worker threads \\
          \texttt{--pattern GLOB}  & File filter (default: \texttt{*.pdf}) \\
        \end{tabular}
      \end{block}
      \vspace{4pt}
      \begin{lstlisting}
# Watermark all PDFs with 4 threads
pdfuse watermark \
  --folder ./contracts/ \
  --text "CONFIDENTIAL" \
  --recursive --workers 4 \
  -o ./output/
      \end{lstlisting}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{block}{Per-Command Flags}
        \small
        \renewcommand{\arraystretch}{1.1}
        \begin{tabular}{ll}
          \multicolumn{2}{l}{\textbf{merge}} \\
          \quad\texttt{--sort name|date} & Sort order \\
          \quad\texttt{--reverse}        & Reverse order \\[2pt]
          \multicolumn{2}{l}{\textbf{split}} \\
          \quad\texttt{--pages N-M}      & Range to extract \\[2pt]
          \multicolumn{2}{l}{\textbf{rotate}} \\
          \quad\texttt{--angle 90|180|270} & Angle (required) \\
          \quad\texttt{--pages N,N,\ldots} & Specific pages \\[2pt]
          \multicolumn{2}{l}{\textbf{watermark}} \\
          \quad\texttt{--text "TEXT"}    & Text overlay \\
          \quad\texttt{--stamp FILE.pdf} & PDF stamp \\[2pt]
          \multicolumn{2}{l}{\textbf{reorder}} \\
          \quad\texttt{--order N,N,N}    & New page sequence \\
        \end{tabular}
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}[fragile]{Demo 1 --- Basic Commands}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \textbf{Merge}
      \begin{lstlisting}
pdfuse merge ch1.pdf ch2.pdf -o book.pdf
# Merged 2 file(s) -> book.pdf
      \end{lstlisting}
      \vspace{2pt}
      \textbf{Split}
      \begin{lstlisting}
pdfuse split report.pdf \
  --pages 3-7 -o excerpt.pdf
# Extracted 5 page(s) -> excerpt.pdf
      \end{lstlisting}
      \vspace{2pt}
      \textbf{Compress}
      \begin{lstlisting}
pdfuse compress report.pdf \
  -o report_small.pdf
# (1204 KB -> 876 KB, 27.2% reduction)
      \end{lstlisting}
    \end{column}
    \begin{column}{0.48\textwidth}
      \textbf{Rotate}
      \begin{lstlisting}
pdfuse rotate scan.pdf \
  --angle 90 --pages 1,3 \
  -o scan_fixed.pdf
      \end{lstlisting}
      \vspace{2pt}
      \textbf{Watermark}
      \begin{lstlisting}
pdfuse watermark contract.pdf \
  --text "CONFIDENTIAL" \
  -o contract_wm.pdf
      \end{lstlisting}
      \vspace{2pt}
      \textbf{Reorder}
      \begin{lstlisting}
pdfuse reorder doc.pdf \
  --order 3,1,2 -o reversed.pdf
# Wrote 3 page(s) in order [3,1,2]
      \end{lstlisting}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}[fragile]{Demo 2 --- Batch Folder Mode}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      Process an \textbf{entire directory} with one command:
      \begin{lstlisting}
pdfuse compress \
  --folder ./drafts/ \
  -o ./compressed/

pdfuse watermark \
  --folder ./contracts/ \
  --recursive \
  --text "DRAFT" \
  -o ./watermarked/ \
  --workers 4
      \end{lstlisting}
      \vspace{4pt}
      Output summary:
      \begin{lstlisting}
Summary: 12 succeeded, 0 failed
      \end{lstlisting}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{block}{Key Design Decisions}
        \begin{itemize}
          \item \textbf{Partial failure} --- one bad file prints a
                warning; processing continues; exit code~1 at the end
          \vspace{3pt}
          \item \textbf{Parallel workers} ---
                \texttt{ThreadPoolExecutor}, configurable via
                \texttt{--workers N}
          \vspace{3pt}
          \item \textbf{Recursive glob} --- preserves relative
                directory structure in output
          \vspace{3pt}
          \item \textbf{Pattern filter} ---
                \texttt{--pattern "report*.pdf"}
        \end{itemize}
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}[fragile]{Demo 3 --- YAML Pipeline: Configuration}
  \begin{columns}[T]
    \begin{column}{0.50\textwidth}
      \textbf{Single-file workflow.yaml}
      \begin{lstlisting}[language={}]
input: report.pdf
steps:
  - compress
  - watermark:
      text: "CONFIDENTIAL"
  - split:
      pages: "1-10"
  - rotate:
      angle: 90
      pages: "2,4"
output: final.pdf
      \end{lstlisting}
      \begin{lstlisting}
pdfuse batch workflow.yaml
      \end{lstlisting}
    \end{column}
    \begin{column}{0.46\textwidth}
      \textbf{Folder pipeline.yaml}
      \begin{lstlisting}[language={}]
input_folder: ./contracts/
pattern: "*.pdf"
steps:
  - compress
  - watermark:
      text: "DRAFT"
output_folder: ./processed/
      \end{lstlisting}
      \vspace{3pt}
      \begin{block}{Available Steps}
        \small
        \renewcommand{\arraystretch}{1.1}
        \begin{tabular}{ll}
          \texttt{compress}   & No parameters \\
          \texttt{watermark}  & \texttt{text} or \texttt{stamp} \\
          \texttt{split}      & \texttt{pages} (required) \\
          \texttt{rotate}     & \texttt{angle}, \texttt{pages} \\
          \texttt{reorder}    & \texttt{order} (required) \\
        \end{tabular}
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Demo 3 --- YAML Pipeline: How It Works}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \begin{block}{Execution Model}
        \begin{enumerate}
          \item Each step reads from a \textbf{temp file},
                writes to the next
          \item Temp directory \textbf{auto-cleaned} on exit
          \item \textbf{Original input is never touched}
          \item On failure: print error, exit(1),
                no partial output written
        \end{enumerate}
      \end{block}
      \vspace{4pt}
      \begin{block}{Folder Mode}
        \begin{itemize}
          \item Same pipeline applied to every matched file
          \item \texttt{pattern} filters which files are processed
          \item Output directory structure mirrors input
        \end{itemize}
      \end{block}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{block}{Why YAML Pipelines?}
        \begin{itemize}
          \item \textbf{Reproducible} --- commit the YAML to version
                control alongside your documents
          \vspace{4pt}
          \item \textbf{No shell scripting} --- chain operations
                without Bash glue code
          \vspace{4pt}
          \item \textbf{Safe} --- temp files are isolated; original
                inputs are never modified
          \vspace{4pt}
          \item \textbf{Portable} --- run the same workflow on any
                machine with \texttt{pdfuse} installed
        \end{itemize}
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Testing --- What Is Tested}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \begin{block}{Happy Path}
        \begin{itemize}
          \item Correct page counts after every operation
          \item Output file exists and is a valid PDF
          \item Folder mode: all files processed, summary correct
        \end{itemize}
      \end{block}
      \vspace{8pt}
      \begin{block}{CLI Contract}
        \begin{itemize}
          \item Mutually exclusive flags
                (\texttt{--text} vs \texttt{--stamp})
          \item Missing required flags: non-zero exit
          \item \texttt{--folder} and FILE are mutually exclusive
        \end{itemize}
      \end{block}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{block}{Edge Cases}
        \begin{itemize}
          \item Out-of-range pages: exit(1), no output file
          \item Empty folder: clean return, exit 0
          \item Failed file in folder: others continue, exit 1
          \item Zero-page PDF, RGBA images, missing files
          \item Original input preserved after batch run
          \item Recursive glob: mirrors directory structure
        \end{itemize}
      \end{block}
    \end{column}
  \end{columns}
\end{frame}
